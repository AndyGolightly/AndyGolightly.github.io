Practical 1 solutions: Metropolis-Hastings

1. Random walk Metropolis. \(N(0,1)\) target.

norm=function(N,a) 
{
 vec = rep(0,N)  # store samples here
 theta = 0 # initial value
 vec[1] = theta # store initial value
 for (i in 2:N) 
 {
  innov = runif(1,-a,a) # generate innovation
  can = theta + innov   # proposal
  laprob = dnorm(can,log=TRUE)-dnorm(theta,log=TRUE) # log of acc. prob.
  u = runif(1)
  if (log(u) < laprob){ 
   theta = can # accept
  }
   vec[i] = theta #store most recent theta
  }
 return(vec) #return vector of samples
}

norm=function(N,a) 
{
 vec = rep(0,N)
 theta = 0
 count = 0
 vec[1] = theta
 for (i in 2:N) 
 {
  innov = runif(1,-a,a)
  can = theta + innov
  laprob = dnorm(can,log=TRUE)-dnorm(theta,log=TRUE)
  u = runif(1)
  if (log(u) < laprob){ 
   theta = can
   count = count+1
  }
   vec[i] = theta
  }
 print(count/N) #empirical overall acceptance rate
 return(vec)
}

2. A simple Metropolis-Hastings independence sampler. \(Gamma(a,b)\) target, normal proposal (with mean and variance to match the target).

gammaMH=function(N,a,b) 
{
 mu = a/b; sig = sqrt(a/(b*b))
 vec = rep(0,N)
 theta = a/b
 vec[1] = theta
 for (i in 2:N) 
 {
  can = rnorm(1,mu,sig)
  laprob1=dgamma(can,a,b,log=TRUE)-dgamma(theta,a,b,log=TRUE)
  laprob2=dnorm(theta,mu,sig,log=TRUE)-dnorm(can,mu,sig,log=TRUE)
  laprob=laprob1+laprob2
  u = runif(1)
  if (log(u) < laprob){ 
   theta = can
  } 
   vec[i] = theta
  }
 return(vec)
}

vec=gammaMH(10000,2.3,2.7)
par(mfrow=c(2,1))
plot(ts(vec))
hist(vec,30)
par(mfrow=c(1,1))

(c) Modify sampler to also return the log of the target density and proposal density at the current value of the chain.

gammaMH=function(N,a,b) 
{
 mu = a/b; sig = sqrt(a/(b*b))
 mat = matrix(0,nrow=N,ncol=3)
 theta = a/b
 mat[1,] = c(theta,dgamma(theta,a,b,log=TRUE),dnorm(theta,mu,sig,log=TRUE))
 for (i in 2:N) 
 {
  can = rnorm(1,mu,sig)
  laprob1=dgamma(can,a,b,log=TRUE)-dgamma(theta,a,b,log=TRUE)
  laprob2=dnorm(theta,mu,sig,log=TRUE)-dnorm(can,mu,sig,log=TRUE)
  laprob=laprob1+laprob2
  u = runif(1)
  if (log(u) < laprob){ 
   theta = can
  } 
   mat[i,] = c(theta,dgamma(theta,a,b,log=TRUE),dnorm(theta,mu,sig,log=TRUE))
  }
 return(mat)
}

mat=gammaMH(10000,0.1,0.01)
#Trace plots
par(mfrow=c(3,1))
plot(ts(mat[,1]),ylab="theta")
plot(ts(mat[,2]),ylab="log pi theta")
plot(ts(mat[,3]),ylab="log q theta")
par(mfrow=c(1,1))

Comment 1: when the chain gets stuck at values of \(\theta\) around 0, the corresponding \(\pi(\theta)\) values are large. (To see this, plot the gamma density for shape values less than 1). Moreover, corresponding \(q(\theta)\) values are relatively small. This leads to the contribution of \(q(\theta) / \pi(\theta)\) in the acceptance probability being very small, in turn leading to \(\alpha(\theta^*|\theta) \approx 0\). Hence the stickiness!

Comment 2: the chain gets stuck at large values of \(\theta\) due to the small values of the corresponding \(q(\theta)\). That is, these values are likely under the target but not the proposal.

rwm=function(N,lambda)
{
 vec = rep(0,N)
 psi = 0
 vec[1] = psi
 for (i in 2:N) 
 {
  innov = rnorm(1,0,lambda)
  can = psi + innov
  laprob = lpost(can)-lpost(psi)
  u = runif(1)
  if (log(u) < laprob){ 
   psi = can
  }
   vec[i] = psi
  }
 return(vec)
}
